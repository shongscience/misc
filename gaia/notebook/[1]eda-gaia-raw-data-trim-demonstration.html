def getHealPixIndexLatLon(inlat, inlon, nside):
    lat_theta = 0.5*np.pi - np.radians(inlat)
    lon_phi = np.radians(inlon)
    #print("lat_theta/pi= ",lat_theta/np.pi)

    return int(hp.ang2pix(nside,theta = lat_theta,phi= lon_phi,nest=True))

import numpy as np
eclgalmat = np.array([ \
                      [-0.054876, -0.873437, -0.483835],\
                      [0.494109, -0.444830, 0.746982],\
                      [-0.867666, -0.198076, 0.455984]\
                     ])
def fast_ecl2gal_deg(radeg,decdeg,rotmat=eclgalmat):
    rarad, decrad = np.radians([radeg,decdeg])
    invec = np.array([np.cos(decrad)*np.cos(rarad),np.cos(decrad)*np.sin(rarad),np.sin(decrad)])
    x0,x1,x2 = np.dot(rotmat,invec)
    
    #python's builit float is double-precision
    return [float(np.degrees(np.arctan2(x1,x0))),float(np.degrees(np.arcsin(x2)))]

def equatorial_to_galactic(right_ascension, declination, \
                           _ra_gal=3.3660329196841534, _dec_gal=0.47347728280415174, _long_ncp=2.1455667597982675):
    ra, dec = np.radians(right_ascension), np.radians(declination)
    
    gal_lat = np.sin(dec) * np.sin(_dec_gal)
    gal_lat = np.arcsin(gal_lat + np.cos(dec) * np.cos(_dec_gal) * np.cos(ra - _ra_gal))
    
    gal_longa = np.cos(dec) * np.sin(ra - _ra_gal) 
    gal_longb = np.sin(dec) * np.cos(_dec_gal)
    gal_longb = gal_longb - np.cos(dec) * np.sin(_dec_gal) * np.cos(ra - _ra_gal)
    gal_long = _long_ncp - np.arctan2(gal_longa,gal_longb)
    
    gal_long, gal_lat = np.degrees([gal_long, gal_lat])
    
    return [float(gal_long), float(gal_lat)]
